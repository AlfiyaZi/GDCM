# JPEG 8 bits project

INCLUDE_DIRECTORIES(BEFORE
  ${CMAKE_CURRENT_BINARY_DIR}
  ${CMAKE_CURRENT_SOURCE_DIR})
ADD_LIBRARY(${JPEG_LIBRARY_NAME}8 ${JPEG_SOURCES})
SET_TARGET_PROPERTIES(${JPEG_LIBRARY_NAME}8 PROPERTIES ${JPEG_LIBRARY_PROPERTIES})
IF(NOT JPEG_INSTALL_NO_LIBRARIES)
  INSTALL(TARGETS ${JPEG_LIBRARY_NAME}8
    RUNTIME DESTINATION ${JPEG_INSTALL_BIN_DIR} COMPONENT Runtime
    LIBRARY DESTINATION ${JPEG_INSTALL_LIB_DIR} COMPONENT Runtime
    ARCHIVE DESTINATION ${JPEG_INSTALL_LIB_DIR} COMPONENT Development
  )
ENDIF(NOT JPEG_INSTALL_NO_LIBRARIES)
